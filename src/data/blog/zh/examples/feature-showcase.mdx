    return fib

print(fibonacci(10))

### 1.3 长代码块自动折叠

超过 15 行的代码块会自动折叠，点击底部按钮展开/收起：

```typescript title="src/plugins/shiki-transformers.ts"
interface HastElement {
  type: "element";
  tagName: string;
  properties: Record<string, unknown>;
  children: (HastElement | HastText)[];
}

interface HastText {
  type: "text";
  value: string;
}

function el(
  tag: string,
  props: Record<string, unknown>,
  children: (HastElement | HastText)[] | string = []
): HastElement {
  return {
    type: "element",
    tagName: tag,
    properties: props,
    children:
      typeof children === "string"
        ? [{ type: "text" as const, value: children }]
        : children,
  };
}

function parseMetaString(str = ""): Record<string, string | true> {
  return Object.fromEntries(
    str.split(" ").reduce(
      (acc: [string, string | true][], cur) => {
        const matched = cur.match(/(.+)?=("(.+)"|'(.+)')$/);
        if (matched === null) return acc;
        const key = matched[1];
        const value = matched[3] || matched[4] || true;
        acc.push([key, value]);
        return acc;
      },
      [] as [string, string | true][]
    )
  );
}
```

<Collapse title="点击展开：Astro 项目结构说明">
  一个典型的 Astro 项目目录结构如下：

  - `src/pages/` — 路由页面
  - `src/layouts/` — 页面布局模板
  - `src/components/` — 可复用组件
  - `src/data/` — 内容集合（博客文章等）
  - `src/styles/` — 全局样式
  - `src/plugins/` — 自定义 Remark/Rehype 插件
  - `public/` — 静态资源
</Collapse>

<Collapse title="点击展开：为什么选择 Astro？">
  Astro 的核心优势：

  1. **零 JS 默认** — 静态页面不发送任何 JavaScript
  2. **岛屿架构** — 仅在需要交互的地方加载 JS
  3. **内容优先** — 原生支持 Markdown/MDX
  4. **框架无关** — 可以混用 React、Vue、Svelte 等
  5. **极致性能** — Lighthouse 接近满分
</Collapse>
