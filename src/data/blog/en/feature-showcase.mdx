### 1.3 Auto-collapse for Long Code Blocks

Code blocks exceeding 15 lines automatically collapse with an expand/collapse toggle:

```typescript title="src/plugins/shiki-transformers.ts"
interface HastElement {
  type: "element";
  tagName: string;
  properties: Record<string, unknown>;
  children: (HastElement | HastText)[];
}

interface HastText {
  type: "text";
  value: string;
}

function el(
  tag: string,
  props: Record<string, unknown>,
  children: (HastElement | HastText)[] | string = []
): HastElement {
  return {
    type: "element",
    tagName: tag,
    properties: props,
    children:
      typeof children === "string"
        ? [{ type: "text" as const, value: children }]
        : children,
  };
}

function parseMetaString(str = ""): Record<string, string | true> {
  return Object.fromEntries(
    str.split(" ").reduce(
      (acc: [string, string | true][], cur) => {
        const matched = cur.match(/(.+)?=("(.+)"|'(.+)')$/);
        if (matched === null) return acc;
        const key = matched[1];
        const value = matched[3] || matched[4] || true;
        acc.push([key, value]);
        return acc;
      },
      [] as [string, string | true][]
    )
  );
}
```

### 8.3 p5.js — Creative Coding

Interactive canvases with theme-aware colors. Move your mouse over the canvas below:

<P5Sketch
  sketch={`
    var trail = [];
    p.setup = function() {
      p.createCanvas(400, 200);
      p.noStroke();
    };
    p.draw = function() {
      var bg = getColor('--background');
      var accent = getColor('--accent');
      p.background(bg);
      trail.push({ x: p.mouseX, y: p.mouseY });
      if (trail.length > 20) trail.shift();
      for (var i = 0; i < trail.length; i++) {
        var size = p.map(i, 0, trail.length, 4, 24);
        var alpha = p.map(i, 0, trail.length, 40, 220);
        p.fill(accent + (p.hex(Math.floor(alpha), 2)));
        p.ellipse(trail[i].x, trail[i].y, size, size);
      }
    };
  `}
/>
