### 1.3 Auto-collapse for Long Code Blocks

Code blocks exceeding 15 lines automatically collapse with an expand/collapse toggle:

```typescript title="src/plugins/shiki-transformers.ts"
interface HastElement {
  type: "element";
  tagName: string;
  properties: Record<string, unknown>;
  children: (HastElement | HastText)[];
}

interface HastText {
  type: "text";
  value: string;
}

function el(
  tag: string,
  props: Record<string, unknown>,
  children: (HastElement | HastText)[] | string = []
): HastElement {
  return {
    type: "element",
    tagName: tag,
    properties: props,
    children:
      typeof children === "string"
        ? [{ type: "text" as const, value: children }]
        : children,
  };
}

function parseMetaString(str = ""): Record<string, string | true> {
  return Object.fromEntries(
    str.split(" ").reduce(
      (acc: [string, string | true][], cur) => {
        const matched = cur.match(/(.+)?=("(.+)"|'(.+)')$/);
        if (matched === null) return acc;
        const key = matched[1];
        const value = matched[3] || matched[4] || true;
        acc.push([key, value]);
        return acc;
      },
      [] as [string, string | true][]
    )
  );
}
```
